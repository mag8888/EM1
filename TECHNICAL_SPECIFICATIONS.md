    }
    
    // –û—Ç–ø–∏—Å–∫–∞ –æ—Ç —Å–æ–±—ã—Ç–∏—è
    off(event, callback) {
        if (this.events.has(event)) {

            const index = callbacks.indexOf(callback);
            if (index > -1) {
                callbacks.splice(index, 1);
            }
        }
    }

            this.events.get(event).forEach(callback => {
                callback(data);
            });
        }
    }
}
